330 def _waitfor(func, pathname, waitall=False):
331 # Perform the operation
332 func(pathname)
333 # Now setup the wait loop
334 if waitall:
335 dirname = pathname
336 else:
337 dirname, name = os.path.split(pathname)
338 dirname = dirname or '.'
339 # Check for `pathname` to be removed from the filesystem.
340 # The exponential backoff of the timeout amounts to a total
341 # of ~1 second after which the deletion is probably an error
342 # anyway.
343 # Testing on an i7@4.3GHz shows that usually only 1 iteration is
344 # required when contention occurs.
345 timeout = 0.001
346 while timeout < 1.0:
347 # Note we are only testing for the existence of the file(s) in
348 # the contents of the directory regardless of any security or
349 # access rights. If we have made it this far, we have sufficient
350 # permissions to do that much using Python's equivalent of the
351 # Windows API FindFirstFile.
352 # Other Windows APIs can fail or give incorrect results when
353 # dealing with files that are pending deletion.
354 L = os.listdir(dirname)
355 if not (L if waitall else name in L):
356 return
357 # Increase the timeout and try again
358 time.sleep(timeout)
359 timeout *= 2
360 warnings.warn('tests may fail, delete still pending for ' + pathname,
361 RuntimeWarning, stacklevel=4)
