103 def guess_type(self, url, strict=True):
104 """Guess the type of a file which is either a URL or a path-like object.
105  
106 Return value is a tuple (type, encoding) where type is None if
107 the type can't be guessed (no or unknown suffix) or a string
108 of the form type/subtype, usable for a MIME Content-type
109 header; and encoding is None for no encoding or the name of
110 the program used to encode (e.g. compress or gzip). The
111 mappings are table driven. Encoding suffixes are case
112 sensitive; type suffixes are first tried case sensitive, then
113 case insensitive.
114  
115 The suffixes .tgz, .taz and .tz (case sensitive!) are all
116 mapped to '.tar.gz'. (This is table-driven too, using the
117 dictionary suffix_map.)
118  
119 Optional `strict' argument when False adds a bunch of commonly found,
120 but non-standard types.
121 """
122 url = os.fspath(url)
123 p = urllib.parse.urlparse(url)
124 if p.scheme and len(p.scheme) > 1:
125 scheme = p.scheme
126 url = p.path
127 else:
128 scheme = None
129 url = os.path.splitdrive(url)[1]
130 if scheme == 'data':
131 # syntax of data URLs:
132 # dataurl := "data:" [ mediatype ] [ ";base64" ] "," data
133 # mediatype := [ type "/" subtype ] *( ";" parameter )
134 # data := *urlchar
135 # parameter := attribute "=" value
136 # type/subtype defaults to "text/plain"
137 comma = url.find(',')
138 if comma < 0:
139 # bad data URL
140 return None, None
141 semi = url.find(';', 0, comma)
142 if semi >= 0:
143 type = url[:semi]
144 else:
145 type = url[:comma]
146 if '=' in type or '/' not in type:
147 type = 'text/plain'
148 return type, None # never compressed, so encoding is None
149 base, ext = posixpath.splitext(url)
150 while (ext_lower := ext.lower()) in self.suffix_map:
151 base, ext = posixpath.splitext(base + self.suffix_map[ext_lower])
152 # encodings_map is case sensitive
153 if ext in self.encodings_map:
154 encoding = self.encodings_map[ext]
155 base, ext = posixpath.splitext(base)
156 else:
157 encoding = None
158 ext = ext.lower()
159 types_map = self.types_map[True]
160 if ext in types_map:
161 return types_map[ext], encoding
162 elif strict:
163 return None, encoding
164 types_map = self.types_map[False]
165 if ext in types_map:
166 return types_map[ext], encoding
167 else:
168 return None, encoding

170 def guess_all_extensions(self, type, strict=True):
171 """Guess the extensions for a file based on its MIME type.
172  
173 Return value is a list of strings giving the possible filename
174 extensions, including the leading dot ('.'). The extension is not
175 guaranteed to have been associated with any particular data stream,
176 but would be mapped to the MIME type `type' by guess_type().
177  
178 Optional `strict' argument when false adds a bunch of commonly found,
179 but non-standard types.
180 """
181 type = type.lower()
182 extensions = list(self.types_map_inv[True].get(type, []))
183 if not strict:
184 for ext in self.types_map_inv[False].get(type, []):
185 if ext not in extensions:
186 extensions.append(ext)
187 return extensions

218 def readfp(self, fp, strict=True):
219 """
220 Read a single mime.types-format file.
221  
222 If strict is true, information will be added to
223 list of standard types, else to the list of non-standard
224 types.
225 """
226 while 1:
227 line = fp.readline()
228 if not line:
229 break
230 words = line.split()
231 for i in range(len(words)):
232 if words[i][0] == '#':
233 del words[i:]
234 break
235 if not words:
236 continue
237 type, suffixes = words[0], words[1:]
238 for suff in suffixes:
239 self.add_type(type, '.' + suff, strict)
