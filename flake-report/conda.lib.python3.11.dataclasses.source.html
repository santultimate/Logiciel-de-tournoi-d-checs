449 def _field_init(f, frozen, globals, self_name, slots):
450 # Return the text of the line in the body of __init__ that will
451 # initialize this field.
452  
453 default_name = f'_dflt_{f.name}'
454 if f.default_factory is not MISSING:
455 if f.init:
456 # This field has a default factory. If a parameter is
457 # given, use it. If not, call the factory.
458 globals[default_name] = f.default_factory
459 value = (f'{default_name}() '
460 f'if {f.name} is _HAS_DEFAULT_FACTORY '
461 f'else {f.name}')
462 else:
463 # This is a field that's not in the __init__ params, but
464 # has a default factory function. It needs to be
465 # initialized here by calling the factory function,
466 # because there's no other way to initialize it.
467  
468 # For a field initialized with a default=defaultvalue, the
469 # class dict just has the default value
470 # (cls.fieldname=defaultvalue). But that won't work for a
471 # default factory, the factory must be called in __init__
472 # and we must assign that to self.fieldname. We can't
473 # fall back to the class dict's value, both because it's
474 # not set, and because it might be different per-class
475 # (which, after all, is why we have a factory function!).
476  
477 globals[default_name] = f.default_factory
478 value = f'{default_name}()'
479 else:
480 # No default factory.
481 if f.init:
482 if f.default is MISSING:
483 # There's no default, just do an assignment.
484 value = f.name
485 elif f.default is not MISSING:
486 globals[default_name] = f.default
487 value = f.name
488 else:
489 # If the class has slots, then initialize this field.

671 # looking in sys.modules (again!), and seems like a waste since

676 # - a_module is the module we want to match
677 # - a_type is the type in that module we want to match

1282 tuples, lists, and dicts.

1293 value = _asdict_inner(getattr(obj, f.name), dict_factory)
1294 result.append((f.name, value))

1297 # obj is a namedtuple. Recurse into it, but the returned
1298 # object is another namedtuple of the same type. This is

1300 # treated (see below), but we just need to create them

1381 def make_dataclass(cls_name, fields, *, bases=(), namespace=None, init=True,
1382 repr=True, eq=True, order=False, unsafe_hash=False,
1383 frozen=False, match_args=True, kw_only=False, slots=False,
1384 weakref_slot=False):
1385 """Return a new dynamically created dataclass.
1386  
1387 The dataclass name will be 'cls_name'. 'fields' is an iterable
1388 of either (name), (name, type) or (name, type, Field) objects. If type is
1389 omitted, use the string 'typing.Any'. Field objects are created by
1390 the equivalent of calling 'field(name, type [, Field-info])'.::
1391  
1392 C = make_dataclass('C', ['x', ('y', int), ('z', int, field(init=False))], bases=(Base,))
1393  
1394 is equivalent to::
1395  
1396 @dataclass
1397 class C(Base):
1398 x: 'typing.Any'
1399 y: int
1400 z: int = field(init=False)
1401  
1402 For the bases and namespace parameters, see the builtin type() function.
1403  
1404 The parameters init, repr, eq, order, unsafe_hash, and frozen are passed to
1405 dataclass().
1406 """
1407  
1408 if namespace is None:
1409 namespace = {}
1410  
1411 # While we're looking through the field names, validate that they
1412 # are identifiers, are not keywords, and not duplicates.
1413 seen = set()
1414 annotations = {}
1415 defaults = {}
1416 for item in fields:
1417 if isinstance(item, str):
1418 name = item
1419 tp = 'typing.Any'
1420 elif len(item) == 2:
1421 name, tp, = item
1422 elif len(item) == 3:
1423 name, tp, spec = item
1424 defaults[name] = spec
1425 else:
1426 raise TypeError(f'Invalid field: {item!r}')
1427  
1428 if not isinstance(name, str) or not name.isidentifier():
1429 raise TypeError(f'Field names must be valid identifiers: {name!r}')
1430 if keyword.iskeyword(name):
1431 raise TypeError(f'Field names must not be keywords: {name!r}')
1432 if name in seen:
1433 raise TypeError(f'Field name duplicated: {name!r}')
1434  
1435 seen.add(name)
1436 annotations[name] = tp
1437  
1438 # Update 'ns' with the user-supplied namespace plus our calculated values.
1439 def exec_body_callback(ns):
1440 ns.update(namespace)
1441 ns.update(defaults)
1442 ns['__annotations__'] = annotations
1443  
1444 # We use `types.new_class()` instead of simply `type()` to allow dynamic creation
1445 # of generic dataclasses.
1446 cls = types.new_class(cls_name, bases, {}, exec_body_callback)
1447  
1448 # Apply the normal decorator.
1449 return dataclass(cls, init=init, repr=repr, eq=eq, order=order,
1450 unsafe_hash=unsafe_hash, frozen=frozen,
1451 match_args=match_args, kw_only=kw_only, slots=slots,
1452 weakref_slot=weakref_slot)

1455 def replace(obj, /, **changes):
1456 """Return a new object replacing specified fields with new values.
1457  
1458 This is especially useful for frozen classes. Example usage::
1459  
1460 @dataclass(frozen=True)
1461 class C:
1462 x: int
1463 y: int
1464  
1465 c = C(1, 2)
1466 c1 = replace(c, x=3)
1467 assert c1.x == 3 and c1.y == 2
1468 """
1469  
1470 # We're going to mutate 'changes', but that's okay because it's a
1471 # new dict, even if called with 'replace(obj, **my_changes)'.
1472  
1473 if not _is_dataclass_instance(obj):
1474 raise TypeError("replace() should be called on dataclass instances")
1475  
1476 # It's an error to have init=False fields in 'changes'.
1477 # If a field is not in 'changes', read its value from the provided obj.
1478  
1479 for f in getattr(obj, _FIELDS).values():
1480 # Only consider normal fields or InitVars.
1481 if f._field_type is _FIELD_CLASSVAR:
1482 continue
1483  
1484 if not f.init:
1485 # Error if this field is specified in changes.
1486 if f.name in changes:
1487 raise ValueError(f'field {f.name} is declared with '
1488 'init=False, it cannot be specified with '
1489 'replace()')
1490 continue
1491  
1492 if f.name not in changes:
1493 if f._field_type is _FIELD_INITVAR and f.default is MISSING:
1494 raise ValueError(f"InitVar {f.name!r} "
1495 'must be specified with replace()')
1496 changes[f.name] = getattr(obj, f.name)
1497  
1498 # Create the new object, which calls __init__() and
1499 # __post_init__() (if defined), using all of the init fields we've
1500 # added and/or left in 'changes'. If there are values supplied in
1501 # changes that aren't fields, this will correctly raise a
1502 # TypeError.
1503 return obj.__class__(**changes)
