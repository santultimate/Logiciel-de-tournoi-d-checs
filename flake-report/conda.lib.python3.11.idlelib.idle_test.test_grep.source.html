123 def report(self, pat):
124 _grep.engine._pat = pat
125 with captured_stdout() as s:
126 _grep.grep_it(re.compile(pat), __file__)
127 lines = s.getvalue().split('\n')
128 lines.pop() # remove bogus '' after last \n
129 return lines
