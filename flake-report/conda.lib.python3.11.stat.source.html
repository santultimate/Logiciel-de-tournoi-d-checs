156 def filemode(mode):
157 """Convert a file's mode to a string of the form '-rwxrwxrwx'."""
158 perm = []
159 for table in _filemode_table:
160 for bit, char in table:
161 if mode & bit == bit:
162 perm.append(char)
163 break
164 else:
165 perm.append("-")
166 return "".join(perm)
