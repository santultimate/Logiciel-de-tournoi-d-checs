52 def _has_surrogates(s):
53 """Return True if s may contain surrogate-escaped binary data."""
54 # This check is based on the fact that unless there are surrogates, utf8
55 # (Python's default encoding) can encode any string. This is the fastest
56 # way to check for surrogates, see bpo-11454 (moved to gh-55663) for timings.
57 try:
58 s.encode()
59 return False
60 except UnicodeEncodeError:
61 return True

65 def _sanitize(string):
66 # Turn any escaped bytes into unicode 'unknown' char. If the escaped
67 # bytes happen to be utf-8 they will instead get decoded, even if they
68 # were invalid in the charset the source was supposed to be in. This
69 # seems like it is not a bad thing; a defect was still registered.
70 original_bytes = string.encode('utf-8', 'surrogateescape')
71 return original_bytes.decode('utf-8', 'replace')

77 def formataddr(pair, charset='utf-8'):
78 """The inverse of parseaddr(), this takes a 2-tuple of the form
79 (realname, email_address) and returns the string value suitable
80 for an RFC 2822 From, To or Cc header.
81  
82 If the first element of pair is false, then the second element is
83 returned unmodified.
84  
85 The optional charset is the character set that is used to encode
86 realname in case realname is not ASCII safe. Can be an instance of str or
87 a Charset-like object which has a header_encode method. Default is
88 'utf-8'.
89 """
90 name, address = pair
91 # The address MUST (per RFC) be ascii, so raise a UnicodeError if it isn't.
92 address.encode('ascii')
93 if name:
94 try:
95 name.encode('ascii')
96 except UnicodeEncodeError:
97 if isinstance(charset, str):
98 charset = Charset(charset)
99 encoded_name = charset.header_encode(name)
100 return "%s <%s>" % (encoded_name, address)
101 else:
102 quotes = ''
103 if specialsre.search(name):
104 quotes = '"'
105 name = escapesre.sub(r'\\\g<0>', name)
106 return '%s%s%s <%s>' % (quotes, name, quotes, address)
107 return address

137 open_pos = pos

152 def getaddresses(fieldvalues, *, strict=True):
153 """Return a list of (REALNAME, EMAIL) or ('','') for each fieldvalue.
154  
155 When parsing fails for a fieldvalue, a 2-tuple of ('', '') is returned in
156 its place.
157  
158 If strict is true, use a strict parser which rejects malformed inputs.
159 """
160  
161 # If strict is true, if the resulting list of parsed addresses is greater
162 # than the number of fieldvalues in the input list, a parsing error has
163 # occurred and consequently a list containing a single empty 2-tuple [('',
164 # '')] is returned in its place. This is done to avoid invalid output.
165 #
166 # Malformed input: getaddresses(['alice@example.com <bob@example.com>'])
167 # Invalid output: [('', 'alice@example.com'), ('', 'bob@example.com')]
168 # Safe output: [('', '')]
169  
170 if not strict:
171 all = COMMASPACE.join(str(v) for v in fieldvalues)
172 a = _AddressList(all)
173 return a.addresslist
174  
175 fieldvalues = [str(v) for v in fieldvalues]
176 fieldvalues = _pre_parse_validation(fieldvalues)
177 addr = COMMASPACE.join(fieldvalues)
178 a = _AddressList(addr)
179 result = _post_parse_validation(a.addresslist)
180  
181 # Treat output as invalid if the number of addresses is not equal to the
182 # expected number of addresses.
183 n = 0
184 for v in fieldvalues:
185 # When a comma is used in the Real Name part it is not a deliminator.
186 # So strip those out before counting the commas.
187 v = _strip_quoted_realnames(v)
188 # Expected number of addresses: 1 + number of commas
189 n += 1 + v.count(',')
190 if len(result) != n:
191 return [('', '')]
192  
193 return result

196 def _check_parenthesis(addr):
197 # Ignore parenthesis in quoted real names.
198 addr = _strip_quoted_realnames(addr)
199  
200 opens = 0
201 for pos, ch in _iter_escaped_chars(addr):
202 if ch == '(':
203 opens += 1
204 elif ch == ')':
205 opens -= 1
206 if opens < 0:
207 return False
208 return (opens == 0)

242 def formatdate(timeval=None, localtime=False, usegmt=False):
243 """Returns a date string as specified by RFC 2822, e.g.:
244  
245 Fri, 09 Nov 2001 01:08:47 -0000
246  
247 Optional timeval if given is a floating point time value as accepted by
248 gmtime() and localtime(), otherwise the current time is used.
249  
250 Optional localtime is a flag that when True, interprets timeval, and
251 returns a date relative to the local timezone instead of UTC, properly
252 taking daylight savings time into account.
253  
254 Optional argument usegmt means that the timezone is written out as
255 an ascii string, not numeric one (so "GMT" instead of "+0000"). This
256 is needed for HTTP, and is only used when localtime==False.
257 """
258 # Note: we cannot use strftime() because that honors the locale and RFC
259 # 2822 requires that day and month names be the English abbreviations.
260 if timeval is None:
261 timeval = time.time()
262 if localtime or usegmt:
263 dt = datetime.datetime.fromtimestamp(timeval, datetime.timezone.utc)
264 else:
265 dt = datetime.datetime.utcfromtimestamp(timeval)
266 if localtime:
267 dt = dt.astimezone()
268 usegmt = False
269 return format_datetime(dt, usegmt)

271 def format_datetime(dt, usegmt=False):
272 """Turn a datetime into a date string as specified in RFC 2822.
273  
274 If usegmt is True, dt must be an aware datetime with an offset of zero. In
275 this case 'GMT' will be rendered instead of the normal +0000 required by
276 RFC2822. This is to support HTTP headers involving date stamps.
277 """
278 now = dt.timetuple()
279 if usegmt:
280 if dt.tzinfo is None or dt.tzinfo != datetime.timezone.utc:
281 raise ValueError("usegmt option requires a UTC datetime")
282 zone = 'GMT'
283 elif dt.tzinfo is None:
284 zone = '-0000'
285 else:
286 zone = dt.strftime("%z")
287 return _format_timetuple_and_zone(now, zone)

355 def unquote(str):
356 """Remove quotes from a string."""
357 if len(str) > 1:
358 if str.startswith('"') and str.endswith('"'):
359 return str[1:-1].replace('\\\\', '\\').replace('\\"', '"')
360 if str.startswith('<') and str.endswith('>'):
361 return str[1:-1]
362 return str

393 def decode_params(params):
394 """Decode parameters list according to RFC 2231.
395  
396 params is a sequence of 2-tuples containing (param name, string value).
397 """
398 new_params = [params[0]]
399 # Map parameter's name to a list of continuations. The values are a
400 # 3-tuple of the continuation number, the string value, and a flag
401 # specifying whether a particular segment is %-encoded.
402 rfc2231_params = {}
403 for name, value in params[1:]:
404 encoded = name.endswith('*')
405 value = unquote(value)
406 mo = rfc2231_continuation.match(name)
407 if mo:
408 name, num = mo.group('name', 'num')
409 if num is not None:
410 num = int(num)
411 rfc2231_params.setdefault(name, []).append((num, value, encoded))
412 else:
413 new_params.append((name, '"%s"' % quote(value)))
414 if rfc2231_params:
415 for name, continuations in rfc2231_params.items():
416 value = []
417 extended = False
418 # Sort by number
419 continuations.sort()
420 # And now append all values in numerical order, converting
421 # %-encodings for the encoded segments. If any of the
422 # continuation names ends in a *, then the entire string, after
423 # decoding segments and concatenating, must have the charset and
424 # language specifiers at the beginning of the string.
425 for num, s, encoded in continuations:
426 if encoded:
427 # Decode as "latin-1", so the characters in s directly
428 # represent the percent-encoded octet values.
429 # collapse_rfc2231_value treats this as an octet sequence.
430 s = urllib.parse.unquote(s, encoding="latin-1")
431 extended = True
432 value.append(s)
433 value = quote(EMPTYSTRING.join(value))
434 if extended:
435 charset, language, value = decode_rfc2231(value)
436 new_params.append((name, (charset, language, '"%s"' % value)))
437 else:
438 new_params.append((name, '"%s"' % value))
439 return new_params

441 def collapse_rfc2231_value(value, errors='replace',
442 fallback_charset='us-ascii'):

444 return unquote(value)

446 # object. We do not want bytes() normal utf-8 decoder, we want a straight

467 def localtime(dt=None, isdst=-1):
468 """Return local time as an aware datetime object.
469  
470 If called without arguments, return current time. Otherwise *dt*
471 argument should be a datetime instance, and it is converted to the
472 local time zone according to the system time zone database. If *dt* is
473 naive (that is, dt.tzinfo is None), it is assumed to be in local time.
474 In this case, a positive or zero value for *isdst* causes localtime to
475 presume initially that summer time (for example, Daylight Saving Time)
476 is or is not (respectively) in effect for the specified time. A
477 negative value for *isdst* causes the localtime() function to attempt
478 to divine whether summer time is in effect for the specified time.
479  
480 """
481 if dt is None:
482 return datetime.datetime.now(datetime.timezone.utc).astimezone()
483 if dt.tzinfo is not None:
484 return dt.astimezone()
485 # We have a naive datetime. Convert to a (localtime) timetuple and pass to
486 # system mktime together with the isdst hint. System mktime will return
487 # seconds since epoch.
488 tm = dt.timetuple()[:-1] + (isdst,)
489 seconds = time.mktime(tm)
490 localtm = time.localtime(seconds)
491 try:
492 delta = datetime.timedelta(seconds=localtm.tm_gmtoff)
493 tz = datetime.timezone(delta, localtm.tm_zone)
494 except AttributeError:
495 # Compute UTC offset and compare with the value implied by tm_isdst.
496 # If the values match, use the zone name implied by tm_isdst.
497 delta = dt - datetime.datetime(*time.gmtime(seconds)[:6])
498 dst = time.daylight and localtm.tm_isdst > 0
499 gmtoff = -(time.altzone if dst else time.timezone)
500 if delta == datetime.timedelta(seconds=gmtoff):
501 tz = datetime.timezone(delta, time.tzname[dst])
502 else:
503 tz = datetime.timezone(delta)
504 return dt.replace(tzinfo=tz)
