553 def parse_config_h(fp, vars=None):
554 """Parse a config.h-style file.
555  
556 A dictionary containing name/value pairs is returned. If an
557 optional dictionary is passed in as the second argument, it is

563 define_rx = re.compile("#define ([A-Z][A-Za-z0-9_]+) (.*)\n")
564 undef_rx = re.compile("/[*] #undef ([A-Z][A-Za-z0-9_]+) [*]/\n")

804 def expand_makefile_vars(s, vars):
805 """Expand Makefile-style variables -- "${foo}" or "$(foo)" -- in
806 'string' according to 'vars' (a dictionary mapping variable names to
807 values). Variables not present in 'vars' are silently expanded to the
808 empty string. The variable values in 'vars' should not contain further
809 variable expansions; if 'vars' is the output of 'parse_makefile()',
810 you're fine. Returns a variable-expanded version of 's'.
811 """
812 import re
813  
814 # This algorithm does multiple expansion, so if vars['foo'] contains
815 # "${bar}", it will expand ${foo} to ${bar}, and then expand
816 # ${bar}... and so forth. This is fine as long as 'vars' comes from
817 # 'parse_makefile()', which takes care of such expansions eagerly,
818 # according to make's variable expansion semantics.
819  
820 while True:
821 m = re.search(_findvar1_rx, s) or re.search(_findvar2_rx, s)
822 if m:
823 (beg, end) = m.span()
824 s = s[0:beg] + vars.get(m.group(1)) + s[end:]
825 else:
826 break
827 return s
